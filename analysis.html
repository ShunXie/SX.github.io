3. t.test : Compare Mean Life Expectancy Between Americas and Europe

extract life expectancy in Americas and Europe

Americas <- data %>% 
  janitor::clean_names() %>% 
  filter(region == "Americas") %>% 
  pull(life_expect)


Europe <- data %>% 
  janitor::clean_names() %>% 
  filter(region == "Europe") %>% 
  pull(life_expect)

test equal variance

var.test(Americas,Europe,alternative = "two.sided",conf.level = 0.95) %>% 
  broom::tidy() %>% 
  knitr::kable()
## Multiple parameters; naming those columns num.df, den.df
